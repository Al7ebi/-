    <script>
        const mainSurahs = [{id:1, name:"الفاتحة"}, {id:2, name:"البقرة"}, {id:18, name:"الكهف"}];
        const juzAmma = [
            {id:78, name:"النبأ"}, {id:79, name:"النازعات"}, {id:80, name:"عبس"}, {id:81, name:"التكوير"},
            {id:82, name:"الإنفطار"}, {id:83, name:"المطففين"}, {id:84, name:"الانشقاق"}, {id:85, name:"البروج"},
            {id:86, name:"الطارق"}, {id:87, name:"الأعلى"}, {id:88, name:"الغاشية"}, {id:89, name:"الفجر"},
            {id:90, name:"البلد"}, {id:91, name:"الشمس"}, {id:92, name:"الليل"}, {id:93, name:"الضحى"},
            {id:94, name:"الشرح"}, {id:95, name:"التين"}, {id:96, name:"العلق"}, {id:97, name:"القدر"},
            {id:98, name:"البينة"}, {id:99, name:"الزلزلة"}, {id:100, name:"العاديات"}, {id:101, name:"القارعة"},
            {id:102, name:"التكاثر"}, {id:103, name:"العصر"}, {id:104, name:"الهمزة"}, {id:105, name:"الفيل"},
            {id:106, name:"قريش"}, {id:107, name:"الماعون"}, {id:108, name:"الكوثر"}, {id:109, name:"الكافرون"},
            {id:110, name:"النصر"}, {id:111, name:"المسد"}, {id:112, name:"الإخلاص"}, {id:113, name:"الفلق"}, {id:114, name:"الناس"}
        ];
        
        const allSurahs = [...mainSurahs, ...juzAmma];
        let tracker = JSON.parse(localStorage.getItem('quran_v2_data')) || {};
        let currentId = null;
        let fontSize = 32;

        function init() {
            const grid = document.getElementById('surah-grid');
            grid.innerHTML = allSurahs.map(s => `
                <button onclick="loadSurah(${s.id})" class="bg-white p-4 rounded-2xl border border-gray-100 shadow-sm text-center active:bg-indigo-50">
                    <span class="block text-indigo-600 font-bold">${s.name}</span>
                </button>
            `).join('');
            updateTracker();
        }

        function updateTracker() {
            const list = document.getElementById('tracker-list');
            list.innerHTML = mainSurahs.map(s => {
                const data = tracker[s.name] || {done: false, count: 0};
                return `
                    <div class="flex justify-between items-center bg-gray-50 p-3 rounded-xl">
                        <span class="font-bold text-gray-700">${s.name}</span>
                        <div class="flex items-center gap-4">
                            <span class="text-xs text-indigo-500 font-bold">${data.count} تكرار</span>
                            <input type="checkbox" ${data.done ? 'checked' : ''} onclick="toggleSurah('${s.name}')" class="w-5 h-5 accent-green-600">
                        </div>
                    </div>
                `;
            }).join('');
            localStorage.setItem('quran_v2_data', JSON.stringify(tracker));
        }

        async function loadSurah(id) {
            currentId = id;
            document.getElementById('home-page').classList.add('hidden');
            document.getElementById('reading-page').classList.remove('hidden');
            document.getElementById('loading').classList.remove('hidden');
            document.getElementById('surah-container').classList.add('hidden');
            
            try {
                const res = await fetch(`https://api.alquran.cloud/v1/surah/${id}`);
                const data = await res.json();
                const surah = data.data;
                const surahInfo = allSurahs.find(s => s.id === id);

                document.getElementById('header-title').innerText = `سُورَةُ ${surahInfo.name}`;
                
                // معالجة البسملة (وضعها في سطر لحالها)
                let bismillah = document.getElementById('bismillah-area');
                let content = document.getElementById('surah-content');
                
                if (id !== 1 && id !== 9) {
                    bismillah.classList.remove('hidden');
                    // إزالة البسملة من أول آية إذا كانت موجودة
                    let firstAyah = surah.ayahs[0].text;
                    const bismillahText = "بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ";
                    if (firstAyah.startsWith(bismillahText)) {
                        surah.ayahs[0].text = firstAyah.replace(bismillahText, "").trim();
                    }
                } else {
                    bismillah.classList.add('hidden');
                }

                content.innerHTML = surah.ayahs.map(a => `${a.text} <span class="text-indigo-400 text-xl mx-1">﴿${a.numberInSurah}﴾</span>`).join(' ');
                
                updateNavButtons(id);
                document.getElementById('mark-btn').onclick = () => finish(surahInfo.name);
                window.scrollTo(0,0);
            } catch (e) { alert("تأكد من اتصالك بالإنترنت"); }
            finally {
                document.getElementById('loading').classList.add('hidden');
                document.getElementById('surah-container').classList.remove('hidden');
            }
        }

        function updateNavButtons(id) {
            const idx = allSurahs.findIndex(s => s.id === id);
            const prev = allSurahs[idx-1];
            const next = allSurahs[idx+1];
            
            const pBtn = document.getElementById('prev-btn');
            const nBtn = document.getElementById('next-btn');

            if(prev) {
                pBtn.style.visibility = "visible";
                document.getElementById('prev-name').innerText = prev.name;
            } else pBtn.style.visibility = "hidden";

            if(next) {
                nBtn.style.visibility = "visible";
                document.getElementById('next-name').innerText = next.name;
            } else nBtn.style.visibility = "hidden";
        }

        function navigate(step) {
            const idx = allSurahs.findIndex(s => s.id === currentId);
            const newSurah = allSurahs[idx + step];
            if(newSurah) loadSurah(newSurah.id);
        }

        function toggleSurah(name) {
            if(!tracker[name]) tracker[name] = {done: false, count: 0};
            tracker[name].done = !tracker[name].done;
            updateTracker();
        }

        function finish(name) {
            if(!tracker[name]) tracker[name] = {done: false, count: 0};
            tracker[name].done = true;
            tracker[name].count++;
            updateTracker();
            alert("تم تسجيل الإنجاز");
            goBack();
        }

        function resetTracker() { if(confirm("تصفير الجدول؟")) { tracker = {}; updateTracker(); } }
        function goBack() { document.getElementById('home-page').classList.remove('hidden'); document.getElementById('reading-page').classList.add('hidden'); }
        function changeFontSize(v) { fontSize += v; document.getElementById('surah-content').style.fontSize = fontSize + 'px'; }

        init();
    </script>
